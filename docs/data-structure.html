6.1 Vector

vector as “a single entity consisting of a collection of things.”

marks = c(23, 67, 98, 34, 98, 21)
print(marks)
## [1] 23 67 98 34 98 21
marks
## [1] 23 67 98 34 98 21
# what are the data types therein
class(marks)
## [1] "numeric"
#check if its a vector 
is.vector(marks)
## [1] TRUE
#vector has no dimensions
dim(marks)
## NULL
# length is used to count number of elements in vectors
length(marks)
## [1] 6
# index is the position of an element in a vector 
# in R it starts at index 1 - lets say we find the third element by index 3
marks[3]
## [1] 98
# access second to fourth values
marks[2:4]
## [1] 67 98 34
# exclude value at specific index we use "-index" for instance 
marks[-2] # excludes value at index 2
## [1] 23 98 34 98 21
# reverse the values of vector 
rev(marks)
## [1] 21 98 34 98 67 23
# vector summary
summary(marks) #finds the summary/descriptive statistics
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##   21.00   25.75   50.50   56.83   90.25   98.00
# ARITHMETIC WITH VECTORS 
## Calculate the total marks 
total_marks = sum(marks)
print(paste("Total marks=",total_marks))
## [1] "Total marks= 341"
## vector multiplication
double_marks =2 * marks 
marks
## [1] 23 67 98 34 98 21
double_marks
## [1]  46 134 196  68 196  42
## vector division 
half_marks = marks/2
half_marks
## [1] 11.5 33.5 49.0 17.0 49.0 10.5
## multiply 2 vectors of equal lenght 
a = c(3, 5, 1)
b = c(7, 3, 9)
ab = a*b
ab
## [1] 21 15  9
ba = b*a # is the same as ab
ba
## [1] 21 15  9
## divide 2 vectors of equal length 
abdiv=a/b
abdiv
## [1] 0.4285714 1.6666667 0.1111111
badiv=b/a
badiv
## [1] 2.333333 0.600000 9.000000
## multiplying vectors of unequal length
e=c(1,2,3,4,5)
f=c(1,2)
ef = e*f #it shows an error

## length
ef #shows results since f is replicated to match e as f=c(1,2,1,2,1)

## [1] 3 5 1

z
